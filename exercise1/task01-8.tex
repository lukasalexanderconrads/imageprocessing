def f(nu, M=256, N=256):
    xx, yy = np.meshgrid(range(M), range(N), indexing='xy')
    return .5 * (np.sin(2 * np.pi * nu * xx / (N - 1) + .5 * np.pi * (N - 1)) + 1) * 255
